<declaration> ::= <type> <identifier>;

<type> ::= int | char | bool

<identifier> ::= <letter> | <identifier> <letter> | <identifier> <digit> | <identifier> <underscore>

<underscore> ::= _

<letter> ::= a | b | c | ... | z | A | B | C | ... | Z

<digit> ::= 0 | 1 | 2 | ... | 9

<assignment> ::= <identifier> = <expression>;

<expression> ::= <term> | <expression> <operator> <term>

<term> ::= <identifier> | <number> | <char>

<operator> ::= + | - | * | / | % | == | != | < | > | <= | >= | && | || | ! | & | | | ^ | ~

<number> ::= <digit> | <number> <digit>

<char> ::= ' <character> '

<character> ::= <letter> | <digit> | <symbol>

<symbol> ::= ! | @ | # | $ | % | ^ | & | * | ( | ) | _ | + | - | = | { | } | [ | ] | | | : | ; | " | < | > | ? | , | . | /

<if> ::= if ( <condition> ) <statement>

<condition> ::= <expression> <operator> <expression> | <condition> <operator> <condition> | ( <condition> )

<while> ::= while ( <condition> ) <statement>

<for> ::= for ( <assignment> ; <condition> ; <assignment> ) <statement>

<statement> ::= <declaration> | <assignment> | <if> | <while> | <for> | { <statement> } | <expression>;
